# üé® Matplotlib - pyplot.scatter() - Wykresy punktowe

## üìö Co to jest pyplot.scatter()?

`matplotlib.pyplot.scatter()` to funkcja do tworzenia wykres√≥w punktowych (scatter plots). To jak gwiazdy na niebie - ka≈ºdy punkt to jedna obserwacja, a razem pokazujƒÖ wzorce! ‚≠ê

## üîß Podstawowa sk≈Çadnia

```python
import matplotlib.pyplot as plt

# Podstawowa sk≈Çadnia
plt.scatter(x, y, s=None, c=None, marker='o', alpha=None, **kwargs)
```

## üíª Podstawowe przyk≈Çady

### 1Ô∏è‚É£ Prosty scatter plot

```python
import matplotlib.pyplot as plt
```

### üìä Analiza korelacji wzrost-waga

```python
# Symulacja danych wzrost-waga
np.random.seed(123)
n_people = 150

# Wzrost (cm) - rozk≈Çad normalny
wzrost = np.random.normal(175, 15, n_people)
wzrost = np.clip(wzrost, 150, 200)  # ograniczenie do realistycznego zakresu

# Waga zale≈ºy od wzrostu + losowo≈õƒá
waga = (wzrost - 100) * 0.8 + np.random.normal(0, 10, n_people)
waga = np.clip(waga, 45, 120)

# P≈Çeƒá wp≈Çywa na rozk≈Çad
plec = np.random.choice(['M', 'K'], n_people)
# Mƒô≈ºczy≈∫ni ≈õrednio wy≈ºsi i ciƒô≈ºsi
wzrost[plec == 'M'] += 8
waga[plec == 'M'] += 5

plt.figure(figsize=(12, 8))

# R√≥≈ºne kolory dla p≈Çci
for p, kolor, label in [('M', 'blue', 'Mƒô≈ºczy≈∫ni'), ('K', 'red', 'Kobiety')]:
    mask = plec == p
    plt.scatter(wzrost[mask], waga[mask], 
               c=kolor, alpha=0.6, s=50, label=label)

plt.xlabel('Wzrost (cm)', fontsize=12)
plt.ylabel('Waga (kg)', fontsize=12)
plt.title('Relacja wzrost-waga w zale≈ºno≈õci od p≈Çci', fontsize=14)
plt.legend()
plt.grid(True, alpha=0.3)

# Dodaj liniƒô trendu
z = np.polyfit(wzrost, waga, 1)  # linia liniowa
p = np.poly1d(z)
plt.plot(wzrost, p(wzrost), "--", color='gray', alpha=0.8, 
         label=f'Trend: y = {z[0]:.2f}x + {z[1]:.1f}')
plt.legend()

plt.tight_layout()
plt.show()

# Korelacja
correlation = np.corrcoef(wzrost, waga)[0, 1]
print(f"Korelacja wzrost-waga: {correlation:.3f}")
```

### üí∞ Analiza cena vs jako≈õƒá produkt√≥w

```python
# Dane produkt√≥w
np.random.seed(456)
n_products = 200

# R√≥≈ºne kategorie produkt√≥w
kategorie = ['Elektronika', 'Odzie≈º', 'Dom', 'Sport']
kategoria = np.random.choice(kategorie, n_products)

# Cena zale≈ºy od kategorii
cena_base = {'Elektronika': 500, 'Odzie≈º': 100, 'Dom': 200, 'Sport': 150}
ceny = []
jakosci = []

for kat in kategoria:
    base_price = cena_base[kat]
    cena = np.random.exponential(base_price/2) + base_price/3
    # Jako≈õƒá koreluje z cenƒÖ, ale z szumem
    jakosc = min(10, max(1, (cena - base_price/3) / (base_price/2) * 5 + 
                            np.random.normal(0, 1.5) + 5))
    ceny.append(cena)
    jakosci.append(jakosc)

ceny = np.array(ceny)
jakosci = np.array(jakosci)

plt.figure(figsize=(14, 10))

# R√≥≈ºne kolory i markery dla kategorii
colors = {'Elektronika': 'blue', 'Odzie≈º': 'red', 'Dom': 'green', 'Sport': 'orange'}
markers = {'Elektronika': 'o', 'Odzie≈º': 's', 'Dom': '^', 'Sport': 'D'}

for kat in kategorie:
    mask = kategoria == kat
    plt.scatter(ceny[mask], jakosci[mask], 
               c=colors[kat], marker=markers[kat], 
               alpha=0.7, s=60, label=kat)

plt.xlabel('Cena (PLN)', fontsize=12)
plt.ylabel('Ocena jako≈õci (1-10)', fontsize=12)
plt.title('Relacja cena-jako≈õƒá r√≥≈ºnych kategorii produkt√≥w', fontsize=14)
plt.legend(title='Kategoria', fontsize=10)
plt.grid(True, alpha=0.3)

# Dodaj linie ≈õrednich dla ka≈ºdej kategorii
for kat in kategorie:
    mask = kategoria == kat
    if np.sum(mask) > 1:
        z = np.polyfit(ceny[mask], jakosci[mask], 1)
        p = np.poly1d(z)
        x_trend = np.linspace(ceny[mask].min(), ceny[mask].max(), 100)
        plt.plot(x_trend, p(x_trend), '--', color=colors[kat], alpha=0.5, linewidth=2)

plt.tight_layout()
plt.show()

# Korelacje per kategoria
print("Korelacja cena-jako≈õƒá per kategoria:")
for kat in kategorie:
    mask = kategoria == kat
    if np.sum(mask) > 1:
        corr = np.corrcoef(ceny[mask], jakosci[mask])[0, 1]
        avg_price = np.mean(ceny[mask])
        avg_quality = np.mean(jakosci[mask])
        print(f"{kat:12}: r={corr:+.3f}, ≈õrednia cena: {avg_price:5.0f} PLN, ≈õrednia jako≈õƒá: {avg_quality:.1f}")
```

### üè† Analiza rynku nieruchomo≈õci

```python
# Symulacja danych o mieszkaniach
np.random.seed(789)
n_apartments = 300

# Powierzchnia mieszka≈Ñ
powierzchnia = np.random.gamma(2, 25)  # rozk≈Çad gamma - realistyczny dla mieszka≈Ñ
powierzchnia = np.clip(powierzchnia, 25, 150)

# Cena zale≈ºy od powierzchni, lokalizacji, wieku
lokalizacja = np.random.choice(['Centrum', '≈ör√≥dmie≈õcie', 'Przedmie≈õcia', 'Peryferie'], n_apartments,
                               p=[0.15, 0.25, 0.35, 0.25])
wiek_budynku = np.random.randint(0, 50, n_apartments)

# Cena za m2 zale≈ºy od lokalizacji
cena_m2_base = {'Centrum': 12000, '≈ör√≥dmie≈õcie': 8000, 'Przedmie≈õcia': 6000, 'Peryferie': 4000}

ceny = []
for i in range(n_apartments):
    lok = lokalizacja[i]
    base = cena_m2_base[lok]
    # Cena maleje z wiekiem budynku
    age_factor = 1 - (wiek_budynku[i] / 100)  # max -50%
    # Dodaj losowo≈õƒá
    noise_factor = np.random.normal(1, 0.2)
    cena_m2 = base * age_factor * noise_factor
    cena_total = cena_m2 * powierzchnia[i]
    ceny.append(cena_total)

ceny = np.array(ceny)

plt.figure(figsize=(15, 10))

# Subplot dla r√≥≈ºnych analiz
plt.subplot(2, 2, 1)
# Powierzchnia vs Cena, kolorowane wed≈Çug lokalizacji
loc_colors = {'Centrum': 'red', '≈ör√≥dmie≈õcie': 'orange', 'Przedmie≈õcia': 'green', 'Peryferie': 'blue'}
for lok in np.unique(lokalizacja):
    mask = lokalizacja == lok
    plt.scatter(powierzchnia[mask], ceny[mask], 
               c=loc_colors[lok], alpha=0.6, s=30, label=lok)
plt.xlabel('Powierzchnia (m¬≤)')
plt.ylabel('Cena (PLN)')
plt.title('Powierzchnia vs Cena')
plt.legend(title='Lokalizacja', fontsize=8)
plt.grid(True, alpha=0.3)

plt.subplot(2, 2, 2)
# Wiek vs Cena za m2, rozmiar = powierzchnia
cena_m2 = ceny / powierzchnia
plt.scatter(wiek_budynku, cena_m2, s=powierzchnia*2, alpha=0.5, c='purple')
plt.xlabel('Wiek budynku (lata)')
plt.ylabel('Cena za m¬≤ (PLN)')
plt.title('Wiek vs Cena/m¬≤ (rozmiar = powierzchnia)')
plt.grid(True, alpha=0.3)

plt.subplot(2, 2, 3)
# Histogram cen wed≈Çug lokalizacji
for lok in np.unique(lokalizacja):
    mask = lokalizacja == lok
    plt.scatter(np.random.normal(0, 0.1, np.sum(mask)), ceny[mask]/1000, 
               c=loc_colors[lok], alpha=0.6, s=20, label=lok)
plt.ylabel('Cena (tys. PLN)')
plt.title('Rozk≈Çad cen wed≈Çug lokalizacji')
plt.xticks([])
plt.legend(fontsize=8)
plt.grid(True, alpha=0.3)

plt.subplot(2, 2, 4)
# Cena vs powierzchnia z regresjƒÖ
plt.scatter(powierzchnia, ceny/1000, alpha=0.5, c='gray', s=20)
# Linia trendu
z = np.polyfit(powierzchnia, ceny/1000, 1)
p = np.poly1d(z)
plt.plot(powierzchnia, p(powierzchnia), "r--", alpha=0.8, linewidth=2)
plt.xlabel('Powierzchnia (m¬≤)')
plt.ylabel('Cena (tys. PLN)')
plt.title(f'Trend: {z[0]:.1f} tys.PLN/m¬≤ + {z[1]:.0f} tys.PLN')
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

# Statystyki
print("=== ANALIZA RYNKU MIESZKA≈É ===")
print(f"Liczba mieszka≈Ñ: {n_apartments}")
print(f"≈örednia powierzchnia: {np.mean(powierzchnia):.1f} m¬≤")
print(f"≈örednia cena: {np.mean(ceny):,.0f} PLN")
print(f"≈örednia cena za m¬≤: {np.mean(cena_m2):,.0f} PLN/m¬≤")

print(f"\nStatystyki wed≈Çug lokalizacji:")
for lok in np.unique(lokalizacja):
    mask = lokalizacja == lok
    avg_price = np.mean(ceny[mask])
    avg_m2 = np.mean(powierzchnia[mask])
    avg_price_m2 = np.mean(cena_m2[mask])
    count = np.sum(mask)
    print(f"{lok:15}: {count:3} mieszka≈Ñ, {avg_m2:4.1f}m¬≤, {avg_price:7.0f} PLN ({avg_price_m2:5.0f} PLN/m¬≤)")
```

### 1Ô∏è‚É£ R√≥≈ºne markery i style

```python
# Demonstracja r√≥≈ºnych marker√≥w
fig, axes = plt.subplots(2, 2, figsize=(12, 10))

# R√≥≈ºne markery
markery = ['o', 's', '^', 'v', '<', '>', 'D', 'p', '*', 'h', '+', 'x']
x_demo = range(len(markery))
y_demo = np.random.rand(len(markery)) * 10

axes[0,0].scatter(x_demo, y_demo, s=100, c=y_demo, cmap='viridis')
for i, marker in enumerate(markery):
    axes[0,0].scatter(i, y_demo[i], s=200, marker=marker, c='red', alpha=0.7)
axes[0,0].set_title('R√≥≈ºne markery')
axes[0,0].set_xticks(x_demo)
axes[0,0].set_xticklabels(markery)

# R√≥≈ºne rozmiary
sizes = [20, 50, 100, 200, 500, 1000]
axes[0,1].scatter(range(len(sizes)), [5]*len(sizes), s=sizes, alpha=0.6)
axes[0,1].set_title('R√≥≈ºne rozmiary')
axes[0,1].set_ylim(0, 10)

# R√≥≈ºne alpha (przezroczysto≈õƒá)
alphas = [0.1, 0.3, 0.5, 0.7, 0.9, 1.0]
for i, alpha in enumerate(alphas):
    axes[1,0].scatter([i], [5], s=200, alpha=alpha, c='blue')
axes[1,0].set_title('R√≥≈ºne alpha (przezroczysto≈õƒá)')
axes[1,0].set_ylim(0, 10)

# Edgecolors i facecolors
axes[1,1].scatter([0, 1, 2], [5, 5, 5], s=200, 
                 facecolors=['red', 'none', 'blue'],
                 edgecolors=['black', 'red', 'black'],
                 linewidth=3)
axes[1,1].set_title('Edge colors i face colors')
axes[1,1].set_ylim(0, 10)

plt.tight_layout()
plt.show()
```

## üìù Podsumowanie

`pyplot.scatter()` to idealne narzƒôdzie do analizy relacji:

- üéØ Pokazuje korelacje miƒôdzy zmiennymi
- üé® Kolory (`c=`) dla trzeciego wymiaru
- üìè Rozmiary (`s=`) dla czwartego wymiaru  
- üîç `alpha=` dla przezroczysto≈õci nak≈ÇadajƒÖcych siƒô punkt√≥w
- üìä Idealne do eksploracji danych i znajdowania wzorc√≥w
- üåà `cmap=` dla map kolor√≥w
- ‚ú® Podstawa dla wielu analiz statystycznych
